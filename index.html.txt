    <div id="nameInputs" class="flex flex-col gap-2"></div>
    <button onclick="startBattle()" class="mt-4 bg-purple-600 hover:bg-purple-800 text-white py-2 px-4 rounded">COME√áAR BATALHA ‚öîÔ∏è</button>

    <div id="groupsOutput" class="grid grid-cols-1 gap-4"></div>
  </div>

  <script>
    let teamSize = 0;

    const warSound = new Howl({
      src: ['https://www.myinstants.com/media/sounds/war-start.mp3'],
      volume: 1.0
    });

    const memeSound = new Howl({
      src: ['https://www.myinstants.com/media/sounds/vish.mp3'],
      volume: 1.0
    });

    function selectMode(size) {
      teamSize = size;
      document.getElementById('nameForm').classList.remove('hidden');
      const totalPlayers = size * 2;
      const nameInputs = document.getElementById('nameInputs');
      nameInputs.innerHTML = '';
      for (let i = 1; i <= totalPlayers; i++) {
        const input = document.createElement('input');
        input.type = 'text';
        input.placeholder = `Lenda ${i}`;
        input.className = 'p-2 rounded bg-gray-800 text-white';
        nameInputs.appendChild(input);
      }
    }

    function shuffle(array) {
      for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
      }
      return array;
    }

    function startBattle() {
      warSound.play();

      const inputs = document.querySelectorAll('#nameInputs input');
      const names = Array.from(inputs).map(input => input.value.trim()).filter(n => n);

      if (names.length !== teamSize * 2) {
        alert(`Voc√™ precisa inserir exatamente ${teamSize * 2} nomes!`);
        memeSound.play();
        return;
      }

      const shuffled = shuffle(names);
      const group1 = shuffled.slice(0, teamSize);
      const group2 = shuffled.slice(teamSize);

      document.getElementById('groupsOutput').innerHTML = `
        <div class="bg-gray-800 p-4 rounded shadow">
          <h3 class="text-xl font-semibold mb-2">Grupo 1 üî•</h3>
          <p>${group1.join(', ')}</p>
        </div>
        <div class="bg-gray-800 p-4 rounded shadow">
          <h3 class="text-xl font-semibold mb-2">Grupo 2 ‚ùÑÔ∏è</h3>
          <p>${group2.join(', ')}</p>
        </div>
      `;

      document.getElementById('result').classList.remove('hidden');
    }
  </script>
